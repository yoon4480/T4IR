## Node.js 모듈

### 간단한 모듈 예제

```javascript
// m1.js

const a = 10;		// stack 영역에 저장
this.b = 20;		// m1.js 객체 영역에 저장
c = 30;					// global 영역에 저장

module.exports = a;		// a의 값만 export
module.exports = {a};	// a의 변수명과 값 모두 export
module.export = b;
```

```javascript
// m2.js

const a = require('./m1');	// 확장명 .js 생략 가능

console.log(a);	// 10
console.log(b);	// b is not defined
console.log(c);	// 10 <-- global 영역에 존재하기 때문에 export하지 않아도 사용 가능
```

```javascript
// m3.js

// function a(b) {
//   console.log(b);
// }
const a = (b) => {	// 화살표 함수로 변환
  console.log(b);
}

module.exports = a;
// module.exports = (b) => ++b;
```

```javascript
// m4.js

const f = require('.m3');

f(10);
```

## Node.js 내장객체

### console

```javascript
const string = 'abc';
const number = 1;
const boolean = true;
const obj = {
    outside : {
        inside : {
            key : 'value',
        },
    },
};

console.time('전체 시간');
// 같은 레이블을 가지는 console.timeEnd와 한 쌍을 이루며, time과 timeEnd 사이의 시간 측정
// timeEnd에서 출력
console.log('평범한 로그. 쉼표로 구분하여 여러 값을 나타낼 수 있음');
console.log(string, number, boolean);
console.error('에러 메시지는 console.error에');
console.dir(obj, {colors : false, depth : 2});
console.dir(obj, {colors : true, depth : 1});
// 객체를 콘솔에 표시
console.time('시간 측정');

for(let i = 0; i < 100000; i++) {
    continue;
}

console.timeEnd('시간 측정');

function b() {
    console.trace('에러 위치 추적');
}

function a() {
    b();
}

a();

console.timeEnd('전체 시간');
```



### 타이머

- global 내장객체 소속
- 주어진 특정 시간에 따라 callback 함수 수행

```javascript
// 단순히 변수 선언만 하는 것처럼 보이지만 할당과 동시에 메소드 호출

const timeout = setTimeout(() => {
    console.log('1.5초 후 실행');
}, 1500);

const interval = setInterval(() => {
    console.log('1초마다 실행');
}, 1000);

const timeout2 = setTimeout(() => {
    console.log('실행되지 않음');
}, 3000);

setTimeout(() => {
    clearTimeout(timeout2);
    clearInterval(interval);
}, 2500);

const immediate = setImmediate(() => {
    console.log("즉시 실행");
});

const immediate2 = setImmediate(() => {
    console.log("실행되지 않음");
});

clearImmediate(immediate2);
```

## http 모듈로 웹 서버 구동

```javascript
const http = require('http');
const server = http.createServer((req, res) => {
    res.writeHead(200, {'Content-Type': 'text/html; charset=utf-8'});
    // 한글이 깨지는 경우, writeHead()로 Character Set 설정
    res.write("<h1>Hello Youngwook</h1>");
    res.end();

});

server.listen(9999, () => {
// 포트번호는 0 ~ 65535 사용 가능. 0 ~ 1024는 주로 시스템 포트로 사용하므로 1025부터 사용하자.
    console.log("Server is ready...");
});

// listen() --> createServer() 순서로 수행
```



- callback 함수로 서버 구성

```javascript
const http = require('http');
const fs = require('fs');

http.createServer((req, res) => {
  	console.log(req.url);	// 요청한 URL 정보
  
    fs.readFile('./index.html', (err, data) => {
      	
        if(err) {   // Error가 존재하면 true
            throw err;
        }
        else {
            res.writeHead(200, {'Content-Type' : 'text/html; charset=utf-8'});
            console.log(data);
            res.end(data);  // readFile()로 읽은 내용을 버퍼에 담아서
        }
    });
}).listen(9999, () => {
    console.log("Server is ready...");
});
```
